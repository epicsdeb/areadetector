00732     double acquirePeriod;

00737     guint64 n_completed_buffers, n_failures, n_underruns;
00738     epicsTimeGetCurrent(&lastFeatureGet);
00739 
00740     /* Loop forever */
00741     while (1) {
00742         /* Wait 5ms for an array to arrive from the queue */
00743         if (epicsMessageQueueReceiveWithTimeout(this->msgQId, &buffer, sizeof(&buffer), 0.005) == -1) {
00744                 /* If no camera, wait for one to appear */
00745                 if (this->camera == NULL || this->connectionValid != 1) {
00746                         continue;
00747                 }
00748                 /* We only want to get a feature once every 25ms (max 40 features/s)
00749                  * so compare timestamp against last feature get
00750                  */
00751                 epicsTimeGetCurrent(&now);
00752                 if (epicsTimeDiffInSeconds(&now, &lastFeatureGet) > 0.025) {
00753                         this->lock();

00755                         if (getFeatures) {
00756                                 this->getNextFeature();
00757                         callParamCallbacks();
00758                         }
00759                         this->unlock();
00760                 }
00761                 continue;
00762         }
00763 
00764                 this->lock();
00765 
00766                 // A small check to ensure that frame callbacks are ignored after the
00767                 // user has pressed STOP.
00768                 // For instance the Photonic Sciences SCMOS camera produce 1-2 frames
00769                 // after calling the arv_camera_stop_acquisition()
00770                 int acquiring = 0;
00771                 getIntegerParam(ADAcquire, &acquiring);
00772                 if (not acquiring) {
00773                     this->freeBufferAndUnlock(buffer);
00774                         asynPrint(this->pasynUserSelf, ASYN_TRACE_FLOW,
00775                                         "%s:%s: Received frame callback after commanding camera stop. Ignoring frame.\n",
00776                                         driverName, functionName);
00777                     continue;
00778                 }
00779 
00780                 /* Get the current parameters */
00781                 getIntegerParam(NDArrayCallbacks, &arrayCallbacks);
00782                 getIntegerParam(NDArrayCounter, &imageCounter);
00783                 getIntegerParam(ADNumImages, &numImages);
00784                 getIntegerParam(ADNumImagesCounter, &numImagesCounter);

00786                 getIntegerParam(NDArrayCallbacks, &arrayCallbacks);
00787                 getDoubleParam(ADAcquirePeriod, &acquirePeriod);
00788                 getIntegerParam(AravisLeftShift, &left_shift);
00789 
00790                 /* The buffer structure does not contain the binning, get that from param lib,
00791                  * but it could be wrong for this frame if recently changed */
00792                 getIntegerParam(ADBinX, &binX);
00793                 getIntegerParam(ADBinY, &binY);
00794                 /* Report a new frame with the counters */
00795                 imageCounter++;
00796                 numImagesCounter++;
00797                 setIntegerParam(NDArrayCounter, imageCounter);

00881                                         for (unsigned int ib = 0; ib < buffer->size / 2; ib++) {
00882                                                 array[ib] = array[ib] << shift;
00883                                         }
00884                                 }
00885                         }
00886                 }
00887 
00888                 if (expected_size != buffer->size) {
00889                         asynPrint(this->pasynUserSelf, ASYN_TRACE_ERROR,

00894                 }

00903                 if (arrayCallbacks) {

00920                 }
00921 
00922                 /* See if acquisition is done */
00923                 if ((imageMode == ADImageSingle) ||
00924                         ((imageMode == ADImageMultiple) &&
00925                          (numImagesCounter >= numImages))) {
00926                         arv_camera_stop_acquisition(this->camera);
00927                         setIntegerParam(ADStatus, ADStatusIdle);
00928                         // Want to make sure we're idle before we callback on ADAcquire
00929                         callParamCallbacks();

00931                         asynPrint(this->pasynUserSelf, ASYN_TRACE_FLOW,

00603                 if (status) setDoubleParam(function, rbv / 1000000);

00610     /* generic feature lookup */

00693     double acquirePeriod;

00698     guint64 n_completed_buffers, n_failures, n_underruns;
00699     epicsTimeGetCurrent(&lastFeatureGet);
00700 
00701     /* Loop forever */
00702     while (1) {
00703         /* Wait 5ms for an array to arrive from the queue */
00704         if (epicsMessageQueueReceiveWithTimeout(this->msgQId, &buffer, sizeof(&buffer), 0.005) == -1) {
00705                 /* If no camera, wait for one to appear */
00706                 if (this->camera == NULL || this->connectionValid != 1) {
00707                         continue;
00708                 }
00709                 /* We only want to get a feature once every 25ms (max 40 features/s)
00710                  * so compare timestamp against last feature get
00711                  */
00712                 epicsTimeGetCurrent(&now);
00713                 if (epicsTimeDiffInSeconds(&now, &lastFeatureGet) > 0.025) {
00714                         this->lock();
00715                         getIntegerParam(AravisGetFeatures, &getFeatures);
00716                         if (getFeatures) {
00717                                 this->getNextFeature();
00718                         callParamCallbacks();
00719                         }
00720                         this->unlock();
00721                 }
00722                 continue;
00723         }
00724 
00725                 this->lock();
00726 
00727                 /* Get the current parameters */
00728                 getIntegerParam(NDArrayCallbacks, &arrayCallbacks);
00729                 getIntegerParam(NDArrayCounter, &imageCounter);
00730                 getIntegerParam(ADNumImages, &numImages);
00731                 getIntegerParam(ADNumImagesCounter, &numImagesCounter);

00740                 getIntegerParam(ADBinY, &binY);
00741                 /* Report a new frame with the counters */

00743                 numImagesCounter++;

00745                 setIntegerParam(ADNumImagesCounter, numImagesCounter);

00747                         setDoubleParam(ADTimeRemaining, (numImages - numImagesCounter) * acquirePeriod);
00748                 }

00751                 if (pRaw == NULL) {
00752                         asynPrint(this->pasynUserSelf, ASYN_TRACE_ERROR,

00754                                         driverName, functionName);

00757                 }

00761                 pRaw->timeStamp = buffer->timestamp_ns / 1.e9;
00762

00771                 pRaw->pAttributeList->add("BayerPattern", "Bayer Pattern", NDAttrInt32, &bayerFormat);
00772                 pRaw->pAttributeList->add("ColorMode", "Color Mode", NDAttrInt32, &colorMode);
00773                 pRaw->dataType = (NDDataType_t) dataType;
00774                 switch (colorMode) {
00775                         case NDColorModeMono:
00776                         case NDColorModeBayer:
00777                                 xDim = 0;
00778                                 yDim = 1;
00779                                 pRaw->ndims = 2;

00784                                 yDim = 2;

00787                                 pRaw->dims[0].offset  = 0;
00788                                 pRaw->dims[0].binning = 1;

00791                         default:
00792                                 asynPrint(this->pasynUserSelf, ASYN_TRACE_ERROR,
00793                                                         "%s:%s: unknown colorMode %d\n",
00794                                                         driverName, functionName, colorMode);
00795                                 this->freeBufferAndUnlock(buffer);
00796                                 continue;
00797                 }
00798                 pRaw->dims[xDim].size    = buffer->width;
00799                 pRaw->dims[xDim].offset  = buffer->x_offset;
00800                 pRaw->dims[xDim].binning = binX;
00801                 pRaw->dims[yDim].size    = buffer->height;
00802                 pRaw->dims[yDim].offset  = buffer->y_offset;
00803                 pRaw->dims[yDim].binning = binY;
00804 
00805                 /* If we are 16 bit, shift by the correct amount */
00806                 if (pRaw->dataType == NDUInt16) {
00807                         expected_size *= 2;
00808                         if (left_shift) {
00809                                 int shift = 0;
00810                                 switch (buffer->pixel_format) {
00811                                         case ARV_PIXEL_FORMAT_MONO_14:

00814                                         case ARV_PIXEL_FORMAT_MONO_12:

00826                                         for (unsigned int ib = 0; ib < buffer->size / 2; ib++) {
00827                                                 array[ib] = array[ib] << shift;
00828                                         }
00829                                 }
00830                         }
00831                 }
00832 
00833                 if (expected_size != buffer->size) {
00834                         asynPrint(this->pasynUserSelf, ASYN_TRACE_ERROR,

00837                         this->freeBufferAndUnlock(buffer);
00838                         continue;
00839                 }
00840         /*
00841                 for (int ib = 0; ib<10; ib++) {
00842                         unsigned char *ix = ((unsigned char *)pRaw->pData) + ib;
00843                         printf("%d,", (int) (*ix));
00844                 }
00845                 printf("\n");
00846         */
00847                 /* Report statistics */
00848                 if (this->stream != NULL) {
00849                         arv_stream_get_statistics(this->stream, &n_completed_buffers, &n_failures, &n_underruns);

00851                         setDoubleParam(AravisFailures, (double) n_failures);
00852                         setDoubleParam(AravisUnderruns, (double) n_underruns);
00853                 }
00854 
00855                 /* See if acquisition is done */
00856                 if ((imageMode == ADImageSingle) ||
00857                         ((imageMode == ADImageMultiple) &&
00858                          (numImagesCounter >= numImages))) {
00859                         arv_camera_stop_acquisition(this->camera);
00860                         setIntegerParam(ADAcquire, 0);
00861                         setIntegerParam(ADStatus, ADStatusIdle);
00862                         asynPrint(this->pasynUserSelf, ASYN_TRACE_FLOW,
00863                                   "%s:%s: acquisition completed\n", driverName, functionName);

00865                 /* Get any attributes that have been defined for this driver */
00866                 this->getAttributes(pRaw->pAttributeList);
00867                 /* Call the callbacks to update any changes */
00868                 callParamCallbacks();
00869                 /* this is a good image, so callback on it */
00870                 if (arrayCallbacks) {
00871                         /* Call the NDArray callback */
00872                         /* Must release the lock here, or we can get into a deadlock, because we can

00874                         this->unlock();

00876                                  "%s:%s: calling imageData callback\n", driverName, functionName);

00881     }

01232 asynStatus aravisCamera::getNextFeature() {
01233         int status = asynSuccess;
01234         const char *featureName;
01235         ArvGcNode *feature;
01236         int *index;
01237         epicsFloat64 floatValue;
01238         epicsInt32 integerValue;
01239         const char *stringValue;
01240     guint64 n_completed_buffers, n_failures, n_underruns;
01241 
01242     /* Get geometry on first run */
01243     if (this->featureKeys == NULL) {
01244         this->featureKeys = g_hash_table_get_keys(this->featureLookup);
01245         this->featureIndex = 0;
01246         return (asynStatus) this->getGeometry();
01247     }
01248 
01249     /* Then iterate through keys */
01250     if (this->featureIndex < g_list_length(this->featureKeys)) {
01251                 index = (int *) g_list_nth_data(this->featureKeys, this->featureIndex);
01252                 featureName = (const char *) g_hash_table_lookup(this->featureLookup, index);
01253                 feature = arv_device_get_feature(this->device, featureName);
01254                 if (feature == NULL) {
01255                         status = asynError;
01256                 } else if (arv_gc_node_get_value_type(feature) == G_TYPE_DOUBLE) {
01257                         floatValue = arv_device_get_float_feature_value (this->device, featureName);
01258                         /* special cases for exposure and frame rate */
01259                         if (*index == ADAcquireTime) floatValue /= 1000000;
01260                         if (*index == ADAcquirePeriod && floatValue > 0) floatValue = 1/floatValue;
01261                         status |= setDoubleParam(*index, floatValue);
01262                 } else if (arv_gc_node_get_value_type(feature) == G_TYPE_STRING) {
01263                         stringValue = arv_device_get_string_feature_value(this->device, featureName);
01264                         status |= setStringParam(*index, stringValue);
01265                 } else {
01266                         integerValue = arv_device_get_integer_feature_value (this->device, featureName);
01267                         if (*index == ADGain) {
01268                                 /* Gain is sometimes an integer */
01269                                 status |= setDoubleParam(*index, integerValue);
01270                         } else {
01271                                 status |= setIntegerParam(*index, integerValue);
01272                         }
01273                 }
01274                 this->featureIndex++;
01275                 return (asynStatus) status;
01276     }
01277 
01278     /* On last tick report statistics */
01279     if (this->stream != NULL) {
01280             arv_stream_get_statistics(this->stream, &n_completed_buffers, &n_failures, &n_underruns);
01281         status |= setDoubleParam(AravisCompleted, (double) n_completed_buffers);
01282         status |= setDoubleParam(AravisFailures, (double) n_failures);
01283         status |= setDoubleParam(AravisUnderruns, (double) n_underruns);
01284         }
01285 
01286     /* ensure we go back to the beginning */
01287     g_list_free(this->featureKeys);
01288     this->featureKeys = NULL;
01289     return (asynStatus) status;

00731 {
00732         int status = asynSuccess;
00733         int acquire;
00734         dc1394error_t err;
00735         epicsUInt32 iBandwidth;
00736         epicsFloat64 dBandwidthPercent;
00737         dc1394video_frame_t * dc1394_frame;
00738         epicsFloat64 fr, timeout;
00739         int stopFromExternalThread = 0; /* The first time, we need this to be 0 so it doesn't signal stop event */
00740         const char *functionName = "imageGrabTask";
00741 
00742         printf("FirewireDCAM::imageGrabTask: Got the image grabbing thread started!\n");
00743         this->lock();
00744         while (1) /* ... round and round and round we go ... */
00745         {
00746                 /* Is acquisition active? */
00747                 getIntegerParam(ADAcquire, &acquire);
00748 
00749                 /* If we are not acquiring then wait for a semaphore that is given when acquisition is started */
00750                 if (!acquire)
00751                 {
00752                         setIntegerParam(ADStatus, ADStatusIdle);
00753                         callParamCallbacks();
00754 
00755                         /* Release the lock while we wait for an event that says acquire has started, then lock again */
00756                         this->unlock();
00757 
00758                         /* Wait for a signal that tells this thread that the transmission
00759                          * has started and we can start asking for image buffers...      */
00760                         if (stopFromExternalThread)     {
00761                                 asynPrint(this->pasynUserSelf, ASYN_TRACE_FLOW,
00762                                                 "%s::%s [%s]: signalling stop event\n", driverName, functionName, this->portName);
00763                                 epicsEventSignal(this->stopEventId);

00767                                         "%s::%s [%s]: waiting for acquire to start\n", driverName, functionName, this->portName);
00768                         /* This is essentially a breakpoint.
00769                          * If someone sets ADAquire to 0, and then waits with 2 second timeout for
00770                          * stopEventId, then this thread will be at this point */
00771                         status = epicsEventWait(this->startEventId);
00772                         asynPrint(this->pasynUserSelf, ASYN_TRACE_FLOW,
00773                                         "%s::%s [%s]: started!\n", driverName, functionName, this->portName);
00774                         this->lock();
00775                 }
00776 
00777                 /* We are now waiting for an image  */
00778                 setIntegerParam(ADStatus, ADStatusWaiting);
00779                 callParamCallbacks();
00780                 getDoubleParam((int)DC1394_FEATURE_FRAME_RATE - (int)DC1394_FEATURE_MIN, FDC_feat_val_abs, &fr);
00781                 if (fr<=0) fr=1;
00782                 timeout = 3./fr;
00783                 err = this->captureDequeueTimeout(this->camera, &dc1394_frame, timeout);
00784                 if (err) {
00785                         /* Frame error */
00786                         status = PERR( this->pasynUserSelf, err );
00787                         continue;
00788                 } else if (dc1394_frame==NULL) {
00789                         /* No frame yet */
00790                         asynPrint(this->pasynUserSelf, ASYN_TRACE_ERROR,
00791                                         "%s::%s [%s]: camera didn't produce frame. Timed out after %.3f s!\n", 
00792                                         driverName, functionName, this->portName, timeout);
00793                         continue;
00794                 }
00795 
00796                 /* We now have a frame, so process it */
00797                 if (this->decodeFrame(dc1394_frame)) {
00798                         /* We abort if we had some problem with grabbing an image...
00799                          * This is perhaps not always the desired behaviour but it'll do for now. */
00800                         this->stopCapture(this->pasynUserSelf);
00801                         stopFromExternalThread = 0;
00802                         continue;
00803                 }
00804 
00805                 /* Get some information about the current feature settings */
00806                 this->getAllFeatures();
00807 
00808                 /* Calculate the bandwidth usage. The 1394 bus has 4915 bandwidth units
00809          * available per cycle. Each unit corresponds to the time it takes to send one
00810          * quadlet at ISO speed S1600. The bandwidth usage at S400 is thus four times the
00811          * number of quadlets per packet.                
00812          */
00813                 dc1394_video_get_bandwidth_usage(this->camera, &iBandwidth);
00814                 dBandwidthPercent = 100.0 * iBandwidth / 4915.0;
00815                 setDoubleParam(FDC_bandwidth, dBandwidthPercent);
00816 
00817         }/* back to the top... */
00818         return;

00867         dc1394color_codings_t codings;

00874         err=dc1394_video_get_supported_modes(this->camera, &video_modes);
00875         ERR( err );

00878          * We only keep the last match we find in the list as this will be the highest resolution. */
00879         for (unsigned int i = 0; i < video_modes.num; i++) {
00880                 /* If the video mode is scalable we break out of the loop
00881                  * because all subsequent modes will be scalable */
00882                 if (video_modes.modes[i] >= DC1394_VIDEO_MODE_FORMAT7_MIN) break;
00883 
00884                 // Check that our demand color coding matches what's available

00891                                                 driverName, functionName, (int)match, (int)colorCoding);
00892                 }
00893         }
00894

00896          * requirements. For each mode we need to investigate all possible colour modes

00900         for(int i = ((int)video_modes.num)-1; i>=0; i--) {
00901                 // If we want to ignore scalables, just break out of the loop
00902                 if (this->disableScalable) break;

00905                 if (video_modes.modes[i] < DC1394_VIDEO_MODE_FORMAT7_MIN) break;

00916                         }

01221     this->pRaw->uniqueId = imageCounter;
01222     this->pRaw->timeStamp = dc1394_frame->timestamp / 1.e6;

01225     this->getAttributes(this->pRaw->pAttributeList);
01226

01228     callParamCallbacks();

01231     if (arrayCallbacks) {
01232         /* Call the NDArray callback */
01233         /* Must release the lock here, or we can get into a deadlock, because we can
01234          * block on the plugin lock, and the plugin can be calling us */
01235         this->unlock();
01236         asynPrint(this->pasynUserSelf, ASYN_TRACE_FLOW,
01237              "%s:%s: calling imageData callback\n", driverName, functionName);
01238         doCallbacksGenericPointer(this->pRaw, NDArrayData, 0);
01239         this->lock();
01240         this->pRaw->release();

01242     }
01243 
01244     /* See if acquisition is done */
01245     if ((imageMode == ADImageSingle) ||
01246         ((imageMode == ADImageMultiple) &&
01247          (numImagesCounter >= numImages))) {
01248         asynPrint(this->pasynUserSelf, ASYN_TRACE_FLOW,
01249               "%s:%s: acquisition completed\n", driverName, functionName);
01250         return asynError;
01251     }
01252 
01253         /* Free the frame off the dc1394 queue. We can do this now as
01254          * we have just taken a local copy of the data.  */
01255         err=dc1394_capture_enqueue(this->camera, dc1394_frame);
01256         status |= PERR( this->pasynUserSelf, err );
01257
